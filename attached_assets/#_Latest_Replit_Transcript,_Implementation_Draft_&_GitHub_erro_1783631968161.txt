```rust
#[no_mangle]
pub extern "C" fn verify_model(
    model_id: u64,
    commitment_hash_ptr: u32, commitment_hash_len: u32,
) -> bool {
    let mut model = match load_model_info(model_id) {
        Some(m) => m,
        None => return false,
    };


    if model.slashed {
        return false;
    }


    let commitment_hash_bytes = read_bytes(commitment_hash_ptr, commitment_hash_len);
    let stored_commitment: ModelCommitment = match serde_json::from_slice(
        &storage_get_value(&model_commitment_key(model_id)).unwrap_or_default()
    ) {
        Ok(c) => c,
        Err(_) => return false,
    };


    // Verify the commitment hash matches the stored one
    if commitment_hash_bytes != sha256(stored_commitment.support_set_hash.as_bytes()).to_vec() {
        return false;
    }


    // Build the verify_residual request
    // The caller must provide the full support data; we'll fetch from IPFS in the host function.
    // We simply pass the model_id and the host function will fetch the support data.
    let req_json = serde_json::to_vec(&serde_json::json!({
        "model_id": model_id,
        "claimed_residual": model.residual,
    })).unwrap();


    let valid = unsafe {
        verify_residual(req_json.as_ptr() as u32, req_json.len() as u32)
    };


    if valid == 1 {
        model.verified = true;
        save_model_info(model_id, &model);
        emit_event(&format!("ModelVerified id={}", model_id));
        true
    } else {
        false
    }
}


#[no_mangle]

        "model_id": model_id,
        "claimed_residual": model.residual,
        "support_data": support_data,

    })).unwrap();


    let valid = unsafe {
        verify_residual(req_json.as_ptr() as u32, req_json.len() as u32)
    };


    let fraudulent = valid != 1;


    if fraudulent {
        // Slash proposer
        let slash_total = model.residual; // example amount
        unsafe {
            slash_account(
                model.proposer.as_bytes().as_ptr(),
                model.proposer.len() as u32,
                &slash_total as *const i64 as u32,
                8,
            );
        }


        // Reward challenger
        let (reward_num, reward_den) = challenge_reward_fraction();
        let reward_amount = (slash_total / reward_den) * reward_num;
        unsafe {
            transfer(
                challenger.as_bytes().as_ptr(),
                challenger.len() as u32,
                &reward_amount as *const i64 as u32,
                8,
            );
        }


        updated_model.slashed = true;
        save_model_info(model_id, &updated_model);
        emit_event(&format!(
            "ModelSlashed id={} challenger={} reward={}",
            model_id, challenger, reward_amount
        ));
        true
    } else {
        // Challenge failed: burn challenger's bond
        let bond = challenge_bond();
        unsafe {
            slash_account(
                challenger.as_bytes().as_ptr(),
                challenger.len() as u32,
                &bond as *const i64 as u32,
                8,
            );
        }
        updated_model.challenged = false;
        save_model_info(model_id, &updated_model);
        emit_event(&format!(
            "ModelChallengeFailed id={} challenger={} bond_burned={}",
            model_id, challenger, bond
        ));
        false
    }
}


#[no_mangle]
pub extern "C" fn get_model(model_id: u64) -> u32 {
    match load_model_info(model_id) {
        Some(info) => {
            let json = serde_json::to_vec(&info).unwrap();
            let ptr = malloc(json.len());
            write_bytes(ptr, &json);
            ptr
        }
        None => 0,
    }
}


// ─── ERC-7992: verifyInference ────────────────────────────────────────────


#[no_mangle]
pub extern "C" fn verify_inference(
    model_id: u64,
    input_commitment_ptr: u32, input_commitment_len: u32,
    output_ptr: u32, output_len: u32,
    proof_ptr: u32, proof_len: u32,
) -> bool {
    // 1. Load model commitment
    let model = match load_model_info(model_id) {
        Some(m) => m,
        None => return false,
    };


    if !model.verified || model.slashed {
        return false;
    }


    let commitment = match serde_json::from_slice::<ModelCommitment>(
        &storage_get_value(&model_commitment_key(model_id)).unwrap_or_default()
    ) {
        Ok(c) => c,
        Err(_) => return false,
    };


    // 2. Get the verifying key for the circuit
    let circuit_id = commitment.circuit_id.as_bytes();
    let mut vk_buf = [0u8; 4096];
    let mut vk_len: u32 = 0;
    let ret = unsafe {
        get_verifying_key(
            circuit_id.as_ptr() as u32,
            circuit_id.len() as u32,
            vk_buf.as_mut_ptr(),
            &mut vk_len as *mut u32,
        )
    };
    if ret != 0 || vk_len == 0 {
        return false;
    }


    // 3. Verify the Groth16 proof
    let input_bytes = read_bytes(input_commitment_ptr, input_commitment_len);
    let output_bytes = read_bytes(output_ptr, output_len);
    let proof_bytes = read_bytes(proof_ptr, proof_len);


    // Public inputs: commitment hash + output
    let mut public_inputs = Vec::new();
    public_inputs.extend_from_slice(&input_bytes);
    public_inputs.extend_from_slice(&output_bytes);


    let valid = unsafe {
        verify_groth16_proof(
            vk_buf.as_ptr() as u32,
            vk_len,
            public_inputs.as_ptr() as u32,
            public_inputs.len() as u32,
            proof_bytes.as_ptr() as u32,
            proof_bytes.len() as u32,
        )
    };


    if valid == 1 {
        emit_event(&format!("InferenceVerified model_id={}", model_id));
        true
    } else {
        false
    }
}
```

```python
import hashlib
from pathlib import Path


def sha256_file(path: Path) -> str:
    """Compute SHA-256 hash of a file and return hex digest."""
    hash_sha = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(4096), b""):
            hash_sha.update(chunk)
    return hash_sha.hexdigest()
```
