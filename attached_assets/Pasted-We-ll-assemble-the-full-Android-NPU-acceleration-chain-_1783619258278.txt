variational-ai/src/bin/export_onnx.rs — full code (as you provided, slightly cleaned):

```rust
use std::fs;
use variational_ai::{
    logistic::LogisticAction,
    mlp::MlpAction,
    mnist::load_synthetic_mnist,
    solver::StationarySolver,
};
use onnx::{
    helper::{make_tensor_value_info, make_node, make_graph, make_model},
    Model, TensorProto,
};

fn export_logistic(path: &str) {
    let data = load_synthetic_mnist(1000);
    let action = LogisticAction::new(data.train_data, data.train_labels, data.dim, 0.01);
    let solver = StationarySolver::new(1e-6, 50);
    let theta = solver.solve_newton_cg(&action, &vec![0.0; data.dim]);

    let input = make_tensor_value_info("input", TensorProto::FLOAT, &[1, data.dim as i64]);
    let output = make_tensor_value_info("output", TensorProto::FLOAT, &[1, 1]);

    let weight_initializer = TensorProto {
        dims: vec![data.dim as i64, 1],
        data_type: TensorProto::FLOAT,
        float_data: theta.iter().map(|&v| v as f32).collect(),
        name: "weights".into(),
        ..Default::default()
    };

    let matmul = make_node("MatMul", &["input", "weights"], &["logits"]);
    let sigmoid = make_node("Sigmoid", &["logits"], &["output"]);

    let graph = make_graph(
        "logistic",
        &[input],
        &[output],
        &[weight_initializer],
        &[matmul, sigmoid],
    );
    let model = make_model(graph);
    fs::write(path, model.encode_to_vec()).unwrap();
    println!("✅ Logistic exported to {}", path);
}

fn export_mlp(path: &str) {
    let data = load_synthetic_mnist(1000);
    let hidden = 32;
    let action = MlpAction::new(data.train_data, data.train_labels, data.dim, hidden, 0.01);
    let solver = StationarySolver::new(1e-5, 80);
    let theta = solver.solve_newton_cg(&action, &vec![0.0; action.param_count()]);
    let (w1, b1, w2, b2) = action.split_params(&theta);

    let input_dim = data.dim as i64;
    let hidden_dim = hidden as i64;

    let input = make_tensor_value_info("input", TensorProto::FLOAT, &[1, input_dim]);
    let output = make_tensor_value_info("output", TensorProto::FLOAT, &[1, 1]);

    let w1_init = TensorProto {
        dims: vec![hidden_dim, input_dim],
        data_type: TensorProto::FLOAT,
        float_data: w1.iter().map(|&v| v as f32).collect(),
        name: "w1".into(),
        ..Default::default()
    };
    let b1_init = TensorProto {
        dims: vec![hidden_dim],
        data_type: TensorProto::FLOAT,
        float_data: b1.iter().map(|&v| v as f32).collect(),
        name: "b1".into(),
        ..Default::default()
    };
    let w2_init = TensorProto {
        dims: vec![1, hidden_dim],
        data_type: TensorProto::FLOAT,
        float_data: w2.iter().map(|&v| v as f32).collect(),
        name: "w2".into(),
        ..Default::default()
    };
    let b2_init = TensorProto {
        dims: vec![1],
        data_type: TensorProto::FLOAT,
        float_data: vec![b2 as f32],
        name: "b2".into(),
        ..Default::default()
    };

    let fc1 = make_node("Gemm", &["input", "w1", "b1"], &["fc1"]);
    let relu = make_node("Relu", &["fc1"], &["relu"]);
    let fc2 = make_node("Gemm", &["relu", "w2", "b2"], &["output"]);

    let graph = make_graph(
        "mlp",
        &[input],
        &[output],
        &[w1_init, b1_init, w2_init, b2_init],
        &[fc1, relu, fc2],
    );
    let model = make_model(graph);
    fs::write(path, model.encode_to_vec()).unwrap();
    println!("✅ MLP exported to {}", path);
}

fn main() {
    let args: Vec<String> = std::env::args().collect();
    if args.len() < 2 {
        eprintln!("Usage: {} <output_dir>", args[0]);
        std::process::exit(1);
    }
    let dir = &args[1];
    fs::create_dir_all(dir).unwrap();
    export_logistic(&format!("{}/logistic.onnx", dir));
    export_mlp(&format!("{}/mlp.onnx", dir));
}
```

3. AI Hub conversion script

scripts/convert_to_hexagon.py — ready-to-use (as provided):

```python
#!/usr/bin/env python3
import argparse
import sys
from pathlib import Path
try:
    from ai_hub import convert, Model
except ImportError:
    print("❌ AI Hub not installed. Run: pip install ai-hub")
    sys.exit(1)

def convert_solver(model_path, output_path, quantization="int8"):
    print(f"Converting {model_path} -> {output_path} ({quantization})")
    try:
        model = Model.from_onnx(model_path)
        result = convert(
            model=model,
            target="hexagon",
            quantization=quantization,
            optimize=["fusion", "sparsity"],
            input_layout="NCHW",
            output_layout="NCHW",
            compiler_options={
                "hexagon": {
                    "vector_extension": "ddp",
                    "cache_blocking": True,
                    "zero_copy": True
                }
            }
        )
        result.save(output_path)
        print(f"✅ Successfully converted: {output_path}")
        return True
    except Exception as e:
        print(f"❌ Conversion failed: {e}")
        return False

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--logistic", type=str)
    parser.add_argument("--mlp", type=str)
    parser.add_argument("--ntk", type=str)        # future use
    parser.add_argument("--output-dir", type=str, default="models/hexagon")
    parser.add_argument("--quantization", type=str, default="int8",
                        choices=["int8", "int4", "mixed"])
    args = parser.parse_args()
    output_dir = Path(args.output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)
    success = True
    if args.logistic:
        success &= convert_solver(args.logistic, str(output_dir / "logistic.dlc"), args.quantization)
    if args.mlp:
        success &= convert_solver(args.mlp, str(output_dir / "mlp.dlc"), args.quantization)
    if success:
        print("\n🎉 All models converted!")
    else:
        print("\n⚠️ Some conversions failed.")
if __name__ == "__main__":
    main()
```

mobile/android/app/src/main/java/com/equilibrium/ai/VariationalAI.kt:

```kotlin
package com.equilibrium.ai

import android.util.Log

object VariationalAI {
    private const val TAG = "VariationalAI"

    init {
        System.loadLibrary("variational_ai")     // libvariational_ai.so
    }

    external fun trainLogistic(): DoubleArray
    external fun trainNtk(): DoubleArray
    external fun solveBlock(blockData: ByteArray): ByteArray

    fun trainLogisticModel(): Double {
        return try {
            val result = trainLogistic()
            if (result.isNotEmpty()) result[0] else 0.0
        } catch (e: Exception) {
            Log.e(TAG, "Logistic training failed", e)
            0.0
        }
    }

    fun trainNtkModel(): Double {
        return try {
            val result = trainNtk()
            if (result.isNotEmpty()) result[0] else 0.0
        } catch (e: Exception) {
            Log.e(TAG, "NTK training failed", e)
            0.0
        }
    }

    fun solveMiningBlock(blockData: ByteArray): ByteArray {
        return try {
            solveBlock(blockData)
        } catch (e: Exception) {
            Log.e(TAG, "Block solving failed", e)
            ByteArray(0)
        }
    }
}
```

mobile/android/app/src/main/java/com/equilibrium/workers/ThermalGuard.kt:

```kotlin
package com.equilibrium.workers

import android.content.Context
import android.os.PowerManager
import android.util.Log

object ThermalGuard {
    private const val TAG = "ThermalGuard"

    fun isThermalSafe(context: Context): Boolean {
        val pm = context.getSystemService(Context.POWER_SERVICE) as PowerManager
        val status = if (android.os.Build.VERSION.SDK_INT >= android.os.Build.VERSION_CODES.Q) {
            pm.currentThermalStatus
        } else {
            PowerManager.THERMAL_STATUS_NONE
        }
        return status == PowerManager.THERMAL_STATUS_NONE ||
               status == PowerManager.THERMAL_STATUS_LIGHT
    }
}
```

mobile/android/app/src/main/java/com/equilibrium/workers/MiningWorker.kt:

```kotlin
package com.equilibrium.workers

import android.content.Context
import android.util.Log
import androidx.work.CoroutineWorker
import androidx.work.WorkerParameters
import com.equilibrium.ai.VariationalAI
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import okhttp3.OkHttpClient
import okhttp3.MediaType.Companion.toMediaType
import okhttp3.Request
import okhttp3.RequestBody.Companion.toRequestBody

class MiningWorker(
    context: Context,
    params: WorkerParameters
) : CoroutineWorker(context, params) {

    private val TAG = "MiningWorker"
    private val client = OkHttpClient()

    override suspend fun doWork(): Result = withContext(Dispatchers.IO) {
        try {
            if (!ThermalGuard.isThermalSafe(applicationContext)) {
                Log.w(TAG, "Thermal throttling active – skipping this cycle")
                return@withContext Result.retry()
            }

            val accuracy = VariationalAI.trainNtkModel()
            Log.i(TAG, "NTK training completed with accuracy: $accuracy")

            val blockData = ByteArray(0)   // TODO: serialize real block header bytes
            val solvedBlock = VariationalAI.solveMiningBlock(blockData)

            if (solvedBlock.isNotEmpty()) {
                submitBlock(solvedBlock)
            }

            Result.success()
        } catch (e: Exception) {
            Log.e(TAG, "Mining worker failed", e)
            Result.retry()
        }
    }

    private fun submitBlock(solvedBlock: ByteArray) {
        try {
            val request = Request.Builder()
                .url("https://your-node/api/blocks/submit")   // replace with your node URL
                .post(solvedBlock.toRequestBody("application/octet-stream".toMediaType()))
                .build()
            client.newCall(request).execute().use { response ->
                if (response.isSuccessful)
                    Log.i(TAG, "Block submitted successfully")
                else
                    Log.e(TAG, "Block submission failed: ${response.code}")
            }
        } catch (e: Exception) {
            Log.e(TAG, "Failed to submit block", e)
        }
    }
}
```

Scheduling the worker (e.g., in MainActivity.onCreate or Application.onCreate):

```kotlin
import androidx.work.*
import java.util.concurrent.TimeUnit

val miningWorkRequest = PeriodicWorkRequestBuilder<MiningWorker>(
    15, TimeUnit.MINUTES
).setConstraints(
    Constraints.Builder()
        .setRequiresCharging(true)
        .setRequiresBatteryNotLow(true)
        .build()
).build()

WorkManager.getInstance(context).enqueueUniquePeriodicWork(
    "equilibrium_mining",
    ExistingPeriodicWorkPolicy.KEEP,
    miningWorkRequest
)
```

5. CI pipeline updates

In .github/workflows/ci.yml or android-apk-ci.yml, add:

```yaml
- name: Export ONNX models
  run: |
    cd variational-ai
    cargo build --release --bin variational-ai-export-onnx
    ./target/release/variational-ai-export-onnx ./models/onnx

- name: Quantize for Hexagon NPU
  run: python3 scripts/convert_to_hexagon.py --logistic models/onnx/logistic.onnx --mlp models/onnx/mlp.onnx --output-dir models/hexagon

- name: Bundle .dlc models with APK
  run: cp models/hexagon/*.dlc app/src/main/assets/
```
